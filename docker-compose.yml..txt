version: '3.8'

services:
  elasticsearch:
    image: docker.elastic.co/elasticsearch/elasticsearch:7.17.10
    container_name: elasticsearch
    environment:
      - discovery.type=single-node  # Single node for simplicity
      - xpack.security.enabled=false  # Disable security for dev
      - ES_JAVA_OPTS=-Xms4g -Xmx4g  # Allocate 4GB RAM
      - cluster.routing.allocation.disk.threshold_enabled=false  # Prevent disk lock
      - bootstrap.memory_lock=true  # Improve performance
    ulimits:
      memlock:
        soft: -1
        hard: -1  # Allow unlimited memory locking
    ports:
      - "9200:9200"  # Expose Elasticsearch port
    volumes:
      - elasticsearch-data:/usr/share/elasticsearch/data  # Persist data
    healthcheck:
      test: ["CMD-SHELL", "curl -s http://localhost:9200/_cluster/health | grep -qE '\"status\":\"(green|yellow)\"'"]
      interval: 30s
      timeout: 10s
      retries: 3

volumes:
  elasticsearch-data:  # Store Elasticsearch data
